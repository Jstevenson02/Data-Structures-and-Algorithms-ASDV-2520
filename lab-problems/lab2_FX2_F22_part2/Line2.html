package events;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.layout.Pane;
import javafx.scene.paint.Color;
import javafx.scene.shape.Line;
import javafx.stage.Stage;

public class Line2 extends Application {

    @Override
    public void start(Stage primaryStage) {
        Scene scene = new Scene(new LinePane2(), 200, 200);
        primaryStage.setTitle("ShowLine");
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}

class LinePane2 extends Pane {

    public LinePane2() {

        Line line2 = new Line(10, 10, 11, 11);
        line2.endXProperty().bind(this.widthProperty().subtract(10));
        line2.endYProperty().bind(this.heightProperty().subtract(10));
        line2.setStrokeWidth(5);
        line2.setStroke(Color.GREEN);
        this.getChildren().add(line2);

        Line line3 = new Line(10, 10, 11, 11);
        line3.startXProperty().bind(this.widthProperty().subtract(10));
        line3.endYProperty().bind(this.heightProperty().subtract(10));
        line3.setStrokeWidth(5);
        line3.setStroke(Color.GREEN);
        this.getChildren().add(line3);

        Line middleLine = new Line();
        middleLine.startXProperty().bind(this.widthProperty().divide(2));
        middleLine.startYProperty().set(0);
        middleLine.endXProperty().bind(this.widthProperty().divide(2));
        middleLine.endYProperty().bind(heightProperty());

        middleLine.setStrokeWidth(10);
        middleLine.setStroke(Color.RED);
        this.getChildren().add(middleLine);
    }
}
